



# Install Venv:

# python3 -m pip install --user virtualenv
#
# On Windows:
# py -m pip install --user virtualenv

# Create venv
# On macOS and Linux:
# python3 -m venv env
#
# On Windows:
# py -m venv env

# Activate Venv
# source env/bin/activate
#
# On Windows:
#
# .\env\Scripts\activate

# Install requirements
# pip install -r requirements.txt
